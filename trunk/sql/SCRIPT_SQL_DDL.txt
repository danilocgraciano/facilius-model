CREATE TABLE usuario(
    id serial not null,
    login varchar(255) UNIQUE not null,
    nome varchar(255) not null,
    foto bytea,
    senha varchar(255) not null,
    email varchar(255) not null,
    tipo integer not null,
    primary key(id)
);

CREATE TABLE usuario_curso(
    matricula serial not null,
    cursoid serial not null,
    usuarioid serial not null,
    primary key(matricula)
);

CREATE TABLE curso(
    id serial not null,
    nome varchar(255) not null,
    primary key(id)
);

CREATE TABLE usuario_curso_turma(
    id serial not null,
    usuario_cursomatricula serial not null,
    turmaid serial not null,
    primary key(id)
);

CREATE TABLE valor_nota(
    id serial not null,
    valor float not null,
    tipo_notaid serial not null,
    usuario_curso_turmaid serial,
    primary key(id)
);

CREATE TABLE tipo_nota(
    id serial not null,
    oficial boolean not null,
    descricao varchar(255) not null,
    primary key(id)
);

CREATE TABLE turma(
    id serial not null,
    descricao varchar(255) not null,
    ano integer not null,
    disciplinaid serial not null,
    professorid serial not null,
    primary key(id)
);

CREATE TABLE frequencia(
    id serial,
    usuario_curso_turmaid serial not null,
    aulaid serial not null,
    status boolean not null,
    primary key(id)
);

CREATE TABLE turma_tipo_nota(
    turmaid serial not null,
    tipo_notaid serial not null,
    primary key(turmaid,tipo_notaid)
);

CREATE TABLE disciplina(
    id serial not null,
    nome varchar(255) not null,
    descricao varchar(255) not null,
    cursoid serial not null,
    primary key(id)
);

CREATE TABLE material(
    id serial not null,
    nome varchar(255) not null,
    descricao varchar(255) ,
    arquivo bytea not null,
    aulaid serial,
    primary key(id)
);

CREATE TABLE aula(
    id serial not null,
    titulo varchar(255) not null,
    descricao varchar(255) not null,
    data date not null,
    turmaid serial not null,
    primary key(id)
);

CREATE TABLE instituicao(
    id serial not null,
    nome varchar(255) not null,
    endereco varchar(255) not null,
    bairro varchar(255) not null,
    numero varchar(255) not null,
    telefone1 varchar(20) not null,
    telefone2 varchar(20),
    cidadeid serial not null,
    logo bytea,
    email varchar(255),
    primary key(id)
);

CREATE TABLE cidade(
    id serial not null,
    descricao varchar(255) not null,
    uf varchar(255) not null,
    primary key(id)
);
